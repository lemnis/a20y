/** @module Radio */

import Command from "./abstract/Command.js";
import getActive from "./../utils/getActive";

/**
 * A checkable input in a group of elements with the same role,
 * only one of which can be checked at a time.
 * 
 * ### Examples
 * 
 * <div role="radio" aria-checked="false" tabindex="0">Apple</div>
 * 
 * ```html
 * <div role="radio" aria-checked="false" tabindex="0">Apple</div>
 * ```
 * 
 * @extends Command
 * @alias module:Radio
 */
class Radio extends Command {
    constructor(...args) {
        super(...args);

        this.addEventListener(
            "key",
            this.onclick.bind(this),
            {key: "space", target: this._node.parentNode}
        );

        this.addEventListener("click", this.onclick.bind(this));
    }

    /**
     * Updates the radio status trough an event
     * 
     * @param {Event} ev
     * @listens click
     * @listens keypress:space
     */
    onclick(ev) {
        if(this._node === ev.target || this === getActive()) {
            if (ev) ev.preventDefault();
            if (typeof super.onclick == "function") super.onclick(ev);

            if (this.disabled !== true) {
                this.checked = true;
            }
        }
    }
    
    /**
     * Updates the checked status
     * 
     * @fires Event:change
     * @fires InputEvent:input
     */
    set checked(value){
        let old = this.checked;

        super.checked = value;

        if(old !== value) {
            this.dispatchEvent(new Event("change", { bubbles: true }));
        }

        this.dispatchEvent(new InputEvent("input"));
    }
}

export default Radio;
